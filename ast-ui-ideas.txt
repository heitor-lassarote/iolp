type Attribute = string * string
type Name = string

type IAST =
    abstract member Generate : ICodegen<'ast> -> Result<string, Error>

type ICodegen<'ast when 'ast :> IAST> =
    abstract member Generate : 'ast -> Result<string, Error>

type Header = Tag | Class | Id

type Css =
    { Header : Header
    ; Attributes : Attribute list
    }

type Csss = Css list

type HtmlAst =
    | Tag of Attribute list * AST list
    | Script of Attribute list * IAST option
    | Text of string
with
    interface IAST with
        this.Generate codegen = codegen this
